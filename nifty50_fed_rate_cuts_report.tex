\section{Data Description}
Dataset: Daily closing prices of the NIFTY 50 Index (\textasciicircum NSEI) from April 2000 to April 2026 --- \textbf{6457} daily observations.

Derived variable: Daily log-return, calculated as:
\begin{equation*}
    r_t = \ln\left( \frac{P_t}{P_{t-1}} \right) \times 100
\end{equation*}

\textbf{Event definition:} FOMC rate cut announcements happen in US evening time (IST next morning). We define the \textbf{event day} as the \textbf{next Indian trading day} after each FOMC rate cut announcement. This captures the immediate market reaction to the rate cut news.

\textbf{Source:}
\begin{itemize}
    \item Investing.com CSV download for NIFTY 50 daily prices (Apr 2000 -- May 2020)
    \item Yahoo Finance (via yfinance Python library) for Jun 2020 -- Apr 2026
    \item Federal Reserve FOMC meeting dates for rate cut event classification
\end{itemize}

\section{Real-World Problem}

\textbf{Research Question:} ``Does the US Federal Reserve interest rate cut have a significant \textbf{positive} effect on the NIFTY 50 index the next trading day?''

\textbf{Hypothesis:} Rate cuts signal easier monetary policy globally, which should boost emerging market equities like NIFTY 50 through capital inflows and improved risk sentiment.

This matters for:
\begin{itemize}
    \item \textbf{Futures traders:} A systematic edge on event days translates to direct profit on NIFTY futures
    \item \textbf{Options traders:} Event-day volatility and direction affect straddle/strangle strategies
    \item \textbf{Portfolio managers:} Adjusting positions ahead of known FOMC rate cut announcements
    \item \textbf{Policy economists:} Quantifying US monetary policy spillover to Indian equities
\end{itemize}

\section{Methodology}

\subsection{Point Estimation}
For each group (event-day vs normal-day), we estimate:
\begin{table}[H]
\centering
\begin{tabularx}{\textwidth}{|l|X|X|}
\hline
\rowcolor{hdrbg} \textcolor{white}{\textbf{Estimator}} & \textcolor{white}{\textbf{Formula}} & \textcolor{white}{\textbf{Justification}} \\
\hline
Sample mean $\bar{x}$ & $\frac{1}{n} \sum x_i$ & Unbiased estimator of $\mu$ \\
\hline
\rowcolor{rowalt}
Sample variance $s^2$ & $\frac{1}{n-1} \sum (x_i - \bar{x})^2$ & Unbiased (Bessel's correction) \\
\hline
Sample proportion $\hat{p}$ & $\hat{p}_E = 16/31 = 0.516$ \newline $\hat{p}_N = 2953/6426 = 0.460$ & MLE for Bernoulli $p$ \\
\hline
\end{tabularx}
\end{table}

\subsection{Interval Estimation}
95\% Confidence Interval for the mean (t-distribution, $\sigma$ unknown):
\begin{equation*}
    \bar{x} \pm t_{\alpha/2, n-1} \cdot \frac{s}{\sqrt{n}}
\end{equation*}

95\% CI for proportion (normal approximation, valid since $n\hat{p} > 5$):
\begin{equation*}
    \hat{p} \pm z_{\alpha/2} \cdot \sqrt{\frac{\hat{p}(1-\hat{p})}{n}}
\end{equation*}

\subsection{Hypothesis Test 1 --- Two-Sample t-Test for Means}
We test whether Fed rate cuts have a positive effect on next-day NIFTY returns.
\begin{equation*}
    H_0: \mu_E = \mu_N \quad \text{vs} \quad H_1: \mu_E > \mu_N \quad \text{(one-tailed, rate cuts boost returns)}
\end{equation*}
Welch's t-statistic (unequal variances assumed):
\begin{equation*}
    t = \frac{\bar{x}_E - \bar{x}_N}{\sqrt{\frac{s_E^2}{n_E} + \frac{s_N^2}{n_N}}}
\end{equation*}
Degrees of freedom via the Welch-Satterthwaite equation. Reject $H_0$ if $p < 0.05$.

\subsection{Hypothesis Test 2 --- z-Test for Difference in Proportions}
We test whether event-days have fewer negative returns (positive effect).
\begin{equation*}
    H_0: p_E = p_N \quad \text{vs} \quad H_1: p_E < p_N \quad \text{(one-tailed, fewer negative days)}
\end{equation*}
\begin{equation*}
    z = \frac{\hat{p}_E - \hat{p}_N}{\sqrt{\hat{p}(1-\hat{p}) \left( \frac{1}{n_E} + \frac{1}{n_N} \right) }}
\end{equation*}
where $\hat{p}$ is the pooled proportion under $H_0$.

\section{Statistical Significance vs Economic Significance}

A critical distinction must be made between \textbf{statistical significance} (can we reject $H_0$ at $\alpha = 0.05$?) and \textbf{economic/practical significance} (is the effect large enough to matter for trading?).

\begin{table}[H]
\centering
\begin{tabularx}{\textwidth}{|l|X|X|}
\hline
\rowcolor{hdrbg} \textcolor{white}{\textbf{Aspect}} & \textcolor{white}{\textbf{Statistical View}} & \textcolor{white}{\textbf{Economic/Trading View}} \\
\hline
Mean spread & $+0.19\%$ per event-day, $p = 0.3614$ & On NIFTY at $\sim$25,000: this is $\sim$48 points per event \\
\hline
\rowcolor{rowalt}
For Futures (1 lot) & Not significant at $\alpha = 0.05$ & Profit $\approx$ \textbf{Rs 1,195} per lot per event \\
\hline
For Options & Sample too small ($n=31$) & ATM calls gain from both delta and IV expansion during event \\
\hline
\rowcolor{rowalt}
Cumulative edge & Wide 95\% CI crosses zero & 31 events $\times$ +0.19\% $=$ \textbf{+5.9\%} cumulative alpha over 25 years \\
\hline
Annualised & CI: [-0.86\%, +1.32\%] & $+0.19\% \times 252 \approx +48.2\%$ annualised edge \\
\hline
\end{tabularx}
\end{table}

\textbf{Why the test fails to reject $H_0$:}
\begin{enumerate}
    \item \textbf{Small sample:} Only $n = 31$ rate cut events in 25 years. The t-test requires large $n$ to detect small effects.
    \item \textbf{Extreme volatility:} Event-day $\sigma = 2.97\%$ ($\sim2.2\times$ normal), widening the CI.
    \item \textbf{Mixed reactions:} Some cuts occur during crises (2008, 2020) where markets fall despite the cut, inflating variance.
\end{enumerate}

\section{Overall Conclusion}

\textbf{Estimation Conclusion:} The data suggests a \textbf{directionally positive but statistically insignificant} effect of Fed rate cuts on NIFTY 50, with \textbf{significantly higher volatility} on event days.

At $\alpha = 0.05$, we do not have sufficient statistical evidence to conclude that:

\begin{enumerate}
    \item Mean NIFTY 50 next-day returns are not significantly higher after Fed rate cuts ($t=+0.36, p=0.3614$). The point estimate is positive ($\bar{x}_E = +0.2322\%$ vs $\bar{x}_N = +0.0410\%$) but the wide CI includes zero.
    \item The probability of a negative return is not significantly fewer on event-days ($z=+0.63, p=0.7359$).
    \item Volatility IS significantly higher on event-days ($F=4.92, p=0.0000$). The null hypothesis of equal variance is \textbf{rejected}.
    \item NIFTY daily returns are non-normal ($JB=35889, p\approx 0$). The Student-t distribution provides a better fit than the Gaussian.
\end{enumerate}

\textbf{Key Insight:} In quantitative finance, a strategy with a positive expected return can be \textbf{economically significant} even without statistical significance at the conventional $\alpha = 0.05$ level. The $+0.19\%$ daily spread, while not rejecting $H_0$, represents a \textbf{tradeable edge} that systematic traders would exploit with proper position sizing and risk management.
